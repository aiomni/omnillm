02. Quick Implementation

Initialize the engine using the OmniRuntime builder for optimized GPU scheduling. 

use omnillm::prelude::*;

#[tokio::main]
async fn main() -> Result<()> {
    // Init with default hardware config
    let engine = OmniEngine::new()
        .with_acceleration(Hardware::Cuda)
        .build()
        .await?;

    // Zero-copy inference
    let response = engine.infer("Analyze trait safety")
        .await?;

    Ok(())
}
